=================================
PTHREADS FUNCTION SYNTAX SHEET
=================================

HEADER FILES
--------------

#include <pthread.h>

=================================
1. pthread_t
=================================

Represents a thread.

Syntax:
pthread_t thread_id;


=================================
2. pthread_create
=================================

Creates a new thread.

Syntax:
pthread_create(&thread_id,
               NULL,
               function_name,
               argument);

Example:
pthread_create(&tid,
               NULL,
               worker,
               NULL);


=================================
3. Thread Function
=================================

Thread execution function.

Syntax:
void* function_name(void *arg)
{
    // thread code

    pthread_exit(NULL);
}

Example:
void* worker(void *arg)
{
    printf("Thread running\n");

    pthread_exit(NULL);
}


=================================
4. pthread_join
=================================

Waits for thread completion.

Syntax:
pthread_join(thread_id,
             NULL);

Example:
pthread_join(tid, NULL);


=================================
5. pthread_exit
=================================

Terminates thread.

Syntax:
pthread_exit(NULL);


=================================
6. pthread_self
=================================

Gets current thread ID.

Syntax:
pthread_self()

Example:
printf("%lu\n", pthread_self());


=================================
7. pthread_equal
=================================

Compares thread IDs.

Syntax:
pthread_equal(thread1,
              thread2);


=================================
8. pthread_mutex_t
=================================

Mutex declaration.

Syntax:
pthread_mutex_t mutex;


=================================
9. pthread_mutex_init
=================================

Initializes mutex.

Syntax:
pthread_mutex_init(&mutex,
                   NULL);


=================================
10. pthread_mutex_lock
=================================

Locks mutex.

Syntax:
pthread_mutex_lock(&mutex);


=================================
11. pthread_mutex_unlock
=================================

Unlocks mutex.

Syntax:
pthread_mutex_unlock(&mutex);


=================================
12. pthread_mutex_destroy
=================================

Destroys mutex.

Syntax:
pthread_mutex_destroy(&mutex);


=================================
13. pthread_cond_t
=================================

Condition variable declaration.

Syntax:
pthread_cond_t cond;


=================================
14. pthread_cond_init
=================================

Initialize condition variable.

Syntax:
pthread_cond_init(&cond,
                  NULL);


=================================
15. pthread_cond_wait
=================================

Wait on condition variable.

Syntax:
pthread_cond_wait(&cond,
                  &mutex);


=================================
16. pthread_cond_signal
=================================

Wake one waiting thread.

Syntax:
pthread_cond_signal(&cond);


=================================
17. pthread_cond_broadcast
=================================

Wake all waiting threads.

Syntax:
pthread_cond_broadcast(&cond);


=================================
18. pthread_cond_destroy
=================================

Destroy condition variable.

Syntax:
pthread_cond_destroy(&cond);


=================================
19. sleep
=================================

Pause thread execution.

Syntax:
sleep(seconds);

Example:
sleep(1);


=================================
20. usleep
=================================

Microsecond sleep.

Syntax:
usleep(microseconds);

Example:
usleep(1000);

=================================

gcc file.c -pthread

#include <stdio.h>
#include <pthread.h>

void* worker(void *arg)
{
    int id = *(int*)arg;

    printf("Thread %d is running\n", id);

    pthread_exit(NULL);
}

int main()
{
    pthread_t threads[4];

    int ids[4];

    for(int i=0; i<4; i++)
    {
        ids[i] = i;

        pthread_create(&threads[i],
                       NULL,
                       worker,
                       &ids[i]);
    }

    {
        pthread_join(threads[i], NULL);
    }

    printf("All threads completed\n");

    return 0;
}










#include <stdio.h>
#include <pthread.h>

int arr[8] = {1,2,3,4,5,6,7,8};

int sum = 0;

pthread_mutex_t mutex;

void* partial_sum(void *arg)
{
    int start = *(int*)arg;

    int local = 0;

    for(int i=start; i<start+4; i++)
    {
        local += arr[i];
    }

    pthread_mutex_lock(&mutex);

    sum += local;

    pthread_mutex_unlock(&mutex);

    pthread_exit(NULL);
}

int main()
{
    pthread_t t1, t2;

    int s1 = 0;
    int s2 = 4;

    pthread_mutex_init(&mutex, NULL);

    pthread_create(&t1, NULL,
                   partial_sum, &s1);

    pthread_create(&t2, NULL,
                   partial_sum, &s2);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    printf("Total Sum = %d\n", sum);

    pthread_mutex_destroy(&mutex);

    return 0;
}










#include <stdio.h>
#include <pthread.h>

#define N 2

int A[N][N] = {{1,2},{3,4}};
int B[N][N] = {{5,6},{7,8}};
int C[N][N];

void* multiply(void *arg)
{
    int row = *(int*)arg;

    for(int j=0; j<N; j++)
    {
        C[row][j] = 0;

        for(int k=0; k<N; k++)
        {
            C[row][j] += A[row][k] * B[k][j];
        }
    }

    pthread_exit(NULL);
}

int main()
{
    pthread_t threads[N];

    int rows[N];

    for(int i=0; i<N; i++)
    {
        rows[i] = i;

        pthread_create(&threads[i],
                       NULL,
                       multiply,
                       &rows[i]);
    }

    for(int i=0; i<N; i++)
    {
        pthread_join(threads[i], NULL);
    }

    printf("Result Matrix:\n");

    for(int i=0; i<N; i++)
    {
        for(int j=0; j<N; j++)
        {
            printf("%d ", C[i][j]);
        }

        printf("\n");
    }

    return 0;
}











#include <stdio.h>
#include <pthread.h>
#include <unistd.h>

int buffer = 0;

pthread_mutex_t mutex;

void* producer(void *arg)
{
    for(int i=1; i<=5; i++)
    {
        pthread_mutex_lock(&mutex);

        buffer = i;

        printf("Produced %d\n", buffer);

        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    pthread_exit(NULL);
}

void* consumer(void *arg)
{
    for(int i=1; i<=5; i++)
    {
        pthread_mutex_lock(&mutex);

        printf("Consumed %d\n", buffer);

        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    pthread_exit(NULL);
}

int main()
{
    pthread_t p, c;

    pthread_mutex_init(&mutex, NULL);

    pthread_create(&p, NULL,
                   producer, NULL);

    pthread_create(&c, NULL,
                   consumer, NULL);

    pthread_join(p, NULL);
    pthread_join(c, NULL);

    pthread_mutex_destroy(&mutex);

    return 0;
}

pthread_mutex_t forks[5];

void* philosopher(void *arg)
{
    int id = *(int*)arg;

    pthread_mutex_lock(&forks[id]);
    pthread_mutex_lock(&forks[(id+1)%5]);

    printf("Philosopher %d eating\n", id);

    sleep(1);

    pthread_mutex_unlock(&forks[id]);
    pthread_mutex_unlock(&forks[(id+1)%5]);

    pthread_exit(NULL);
}

int main()
{
    pthread_t phil[5];

    int ids[5];

    for(int i=0; i<5; i++)
    {
        pthread_mutex_init(&forks[i], NULL);
    }

    for(int i=0; i<5; i++)
    {
        ids[i] = i;

        pthread_create(&phil[i],
                       NULL,
                       philosopher,
                       &ids[i]);
    }

    for(int i=0; i<5; i++)
    {
        pthread_join(phil[i], NULL);
    }

    return 0;
}










#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define BUFFER_SIZE 5
#define NUM_PRODUCERS 3
#define NUM_CONSUMERS 3
#define ITEMS_PER_PRODUCER 5

int buffer[BUFFER_SIZE];

int in = 0;
int out = 0;
int count = 0;

pthread_mutex_t mutex;

pthread_cond_t empty;
pthread_cond_t full;


// PRODUCER
void* producer(void *arg)
{
    int id = *(int*)arg;

    for(int i = 1; i <= ITEMS_PER_PRODUCER; i++)
    {
        int item = id * 100 + i;

        pthread_mutex_lock(&mutex);

        // Wait if buffer full
        while(count == BUFFER_SIZE)
        {
            pthread_cond_wait(&empty, &mutex);
        }

        // Produce item
        buffer[in] = item;

        printf("Producer %d produced %d at %d\n",
               id, item, in);

        in = (in + 1) % BUFFER_SIZE;

        count++;

        // Signal consumers
        pthread_cond_signal(&full);

        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    pthread_exit(NULL);
}


// CONSUMER
void* consumer(void *arg)
{
    int id = *(int*)arg;

    for(int i = 1; i <= ITEMS_PER_PRODUCER; i++)
    {
        pthread_mutex_lock(&mutex);

        // Wait if buffer empty
        while(count == 0)
        {
            pthread_cond_wait(&full, &mutex);
        }

        // Consume item
        int item = buffer[out];

        printf("Consumer %d consumed %d from %d\n",
               id, item, out);

        out = (out + 1) % BUFFER_SIZE;

        count--;

        // Signal producers
        pthread_cond_signal(&empty);

        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    pthread_exit(NULL);
}


int main()
{
    pthread_t prod[NUM_PRODUCERS];
    pthread_t cons[NUM_CONSUMERS];

    int prod_id[NUM_PRODUCERS];
    int cons_id[NUM_CONSUMERS];

    // Initialize mutex
    pthread_mutex_init(&mutex, NULL);

    // Initialize condition variables
    pthread_cond_init(&empty, NULL);
    pthread_cond_init(&full, NULL);

    // Create producer threads
    for(int i = 0; i < NUM_PRODUCERS; i++)
    {
        prod_id[i] = i + 1;

        pthread_create(&prod[i],
                       NULL,
                       producer,
                       &prod_id[i]);
    }

    // Create consumer threads
    for(int i = 0; i < NUM_CONSUMERS; i++)
    {
        cons_id[i] = i + 1;

        pthread_create(&cons[i],
                       NULL,
                       consumer,
                       &cons_id[i]);
    }

    // Join producers
    for(int i = 0; i < NUM_PRODUCERS; i++)
    {
        pthread_join(prod[i], NULL);
    }

    // Join consumers
    for(int i = 0; i < NUM_CONSUMERS; i++)
    {
        pthread_join(cons[i], NULL);
    }

    // Destroy mutex and condition variables
    pthread_mutex_destroy(&mutex);

    pthread_cond_destroy(&empty);
    pthread_cond_destroy(&full);

    return 0;
}
