//Secs
#include <stdio.h>
#include <omp.h>

double computeA() {
    return 10.0;
}

double computeB() {
    return 20.0;
}

int main() {
    double sharedVar = 0.0;
    double finalResult = 0.0;

    #pragma omp parallel shared(sharedVar) reduction(+:finalResult)
    {
        int tid = omp_get_thread_num();

        #pragma omp single
        {
            sharedVar = 5.0;
            printf("Thread %d initialized sharedVar = %.2f\n", tid, sharedVar);
        }

        #pragma omp sections
        {
            #pragma omp section
            {
                double resultA = computeA() * sharedVar;
                printf("Thread %d computed Section A: %.2f\n", tid, resultA);
                finalResult += resultA;
            }

            #pragma omp section
            {
                double resultB = computeB() * sharedVar;
                printf("Thread %d computed Section B: %.2f\n", tid, resultB);
                finalResult += resultB;
            }
        }
    }

    printf("Final Reduced Result = %.2f\n", finalResult);
    return 0;
}





//matvecmult
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define SIZE 50

double privateVar;
#pragma omp threadprivate(privateVar)

int main() {
    long mat[SIZE][SIZE], vec[SIZE], result[SIZE];
    long sum = 0.0;
    long max = 0;

    for(int i = 0; i < SIZE; i++) {
        vec[i] = rand() % 10 +1;
        for(int j = 0; j < SIZE; j++)
            mat[i][j] = rand() % 10 + 1;
    }

    #pragma omp parallel
    {
        privateVar = omp_get_thread_num();

        #pragma omp single
        {
            printf("Matrix-vector multiplication started...\n\n");
        }

        #pragma omp for reduction(+:result[:SIZE])
        for(int i = 0; i < SIZE; i++) {
            result[i] = 0;
            for(int j = 0; j < SIZE; j++)
                result[i] += mat[i][j] * vec[j];
        }

        #pragma omp sections
        {
            #pragma omp section
            {
                for(int i = 0; i < SIZE; i++)
                    sum += result[i];
            }

            #pragma omp section
            {
                for(int i = 0; i < SIZE; i++)
                    if(result[i] > max)
                        max = result[i];
            }
        }
    }
	
	printf("Resultant Vector:\n");
	for(int i = 0; i < SIZE; i++) 
	{	
        printf("%ld	",result[i]);
        if((i+1)%10==0){printf("\n");}
    }
	
    printf("\nSum of result vector = %ld\n", sum);
    printf("Maximum value in result vector = %ld\n", max);

    return 0;
}




//nestdot
#include <stdio.h>
#include <stdlib.h>
#include <omp.h>
#include <math.h>	

#define N 1000

int main() {
    int i;
    double a[N], b[N];

    srand(67);
    for (i = 0; i < N; i++) {
        a[i] = (rand() % 100) + 1;
        b[i] = (rand() % 100) + 1;
    }

    long serial_dot = 0;
    for (i = 0; i < N; i++) {
        serial_dot += a[i] * b[i];
    }

    omp_set_nested(1);

    long parallel_dot = 0;

    #pragma omp parallel
    {
        int tid = omp_get_thread_num();
        int nthreads = omp_get_num_threads();

        int chunk = N / nthreads;
        int start = tid * chunk;
        int end = (tid == nthreads - 1) ? N : start + chunk;

        double local_sum = 0.0;

        #pragma omp parallel for reduction(+:local_sum)
        for (int i = start; i < end; i++) {
            local_sum += a[i] * b[i];
        }

        #pragma omp atomic
        parallel_dot += local_sum;
    }

    printf("Serial dot product   = %ld\n", serial_dot);
    printf("Parallel dot product = %ld\n", parallel_dot);

    if (serial_dot==parallel_dot) {
        printf("Result is Right\n");
    } else {
        printf("Result is Wrong \n");
    }

    return 0;
}



//lock
#include <stdio.h>
#include <omp.h>

int main() {

    int value = 0;
    omp_lock_t lock;

    omp_init_lock(&lock);

#pragma omp parallel
{
    omp_set_lock(&lock);

    value += 1;

    omp_unset_lock(&lock);
}

    omp_destroy_lock(&lock);

    printf("value = %d\n", value);
}

Lock functions:

Function	Meaning
omp_init_lock()	create lock
omp_set_lock()	acquire lock
omp_unset_lock()	release lock
omp_destroy_lock()	delete lock




//fibtask
int done[100];
int value[100];

int fib(int n) {

    if(n < 2)
        return n;

    if(!done[n]) {

        int i, j;

#pragma omp task shared(i)
        i = fib(n-1);

#pragma omp task shared(j)
        j = fib(n-2);

#pragma omp taskwait

        value[n] = i + j;
        done[n] = 1;
    }

    return value[n];
}





//deadlock
#include <stdio.h>
#include <omp.h>

int main() {

    omp_lock_t lockA, lockB;
    int a = 0, b = 0;

    omp_init_lock(&lockA);
    omp_init_lock(&lockB);

#pragma omp parallel sections
{
#pragma omp section
{
    omp_set_lock(&lockA);
    printf("Thread 1 locked A\n");

    omp_set_lock(&lockB);   // waits if thread 2 has B
    printf("Thread 1 locked B\n");

    a++;
    b++;

    omp_unset_lock(&lockB);
    omp_unset_lock(&lockA);
}

#pragma omp section
{
    omp_set_lock(&lockB);
    printf("Thread 2 locked B\n");

    omp_set_lock(&lockA);   // waits if thread 1 has A
    printf("Thread 2 locked A\n");

    a++;
    b++;

    omp_unset_lock(&lockA);
    omp_unset_lock(&lockB);
}
}

    omp_destroy_lock(&lockA);
    omp_destroy_lock(&lockB);

    return 0;
}
